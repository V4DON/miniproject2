create table "TrainingPlace" (
id serial primary key,
full_name varchar(150),
short_name varchar(50)
)

create type training_type as enum ('Повышение квалификации', 'Переподготовка', 'Новое образование')

create type training_format as enum ('Очное', 'Дистанционное', 'Гибридное')

create table "Training" (
id serial primary key,
name varchar(150),
type training_type,
start_date date,
end_date date,
format training_format,
training_place_id int references "TrainingPlace"(id)
)

create table "Position" (
id serial primary key,
name varchar(50),
job_description varchar(100)
)

create table "Document" (
id serial primary key,
series int,
number int,
issue_date date,
issued_by varchar(50)
)

create type marital_status as enum ('Холост/Не замужем', 'Женат/За мужем', 'Разведён/Разведена', 'Вдова/Вдовец')

create table "Employee" (
id serial primary key,
last_name varchar(50),
first_name varchar(50),
middle_name varchar(50),
phone_number varchar(20),
birth_date date,
snils varchar(11),
inn varchar(10),
passport_id int references "Document"(id),
work_experience_years int,
marital_status marital_status ,
employment_date date,
termination_date date,
delete boolean
)


create type department_type as enum ('Отдел продаж', 'Бухгалтерия', 'IT отдел')

create table "EmployeePosition" (
id serial primary key,
position_id int references "Position"(id),
employee_id int references "Employee"(id),
department department_type
)

create table "EmployeeTraining" (
id serial primary key,
training_id int references "Training"(id),
employee_id int references "Employee"(id),
training_completed boolean,
document_number varchar(50)
)

create table "Qualification" (
id serial primary key,
name varchar(50),
description varchar(100)
)

create table "Specialty" (
id serial primary key,
full_name varchar(150),
short_name varchar(50),
qualification_id int references "Qualification"(id)
)
 
create type education_level as enum('Среднее профессиональное', 'Высшее')

create table "Education" (
id serial primary key,
level education_level,
series int,
number int,
registration_number int,
issue_date date,
specialty_id int references "Specialty"(id)
)

create table "EmployeeEducation" (
id serial primary key,
employee_id int references "Employee"(ID),
education_id int references "Education"(id)
)
